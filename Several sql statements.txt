User table:

CREATE TABLE USER (USERNAME VARCHAR(100),
                   CREATE_USER INTEGER NOT NULL REFERENCES USER(USERNAME),
                   CREATE_DATE DATE NOT NULL,
                   UPDATE_USER INTEGER REFERENCES USER(USERNAME),
                   UPDATE_DATE DATE,
                   PASSWORD VARCHAR(100) NOT NULL,
                   FIRST_NAME VARCHAR(100) NOT NULL,
                   LAST_NAME VARCHAR(100) NOT NULL,
                   EMAIL VARCHAR(100) NOT NULL,
                   PHONE VARCHAR(100),
                   ENABLED BOOLEAN,
                   PRIMARY KEY(USERNAME)
                  )
				  
				  
				 
user Rollen:
				 
CREATE TABLE USER_USER_ROLE (
    USER_USERNAME VARCHAR(100) REFERENCES USER(USERNAME),
    ROLES VARCHAR(100),
    PRIMARY KEY (USER_USERNAME, ROLES)
    )
